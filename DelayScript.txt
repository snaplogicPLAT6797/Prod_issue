from com.snaplogic.scripting.language import ScriptHook
from com.snaplogic.scripting.language.ScriptHook import *

import time

class DelayScript(ScriptHook):
    def __init__(self, input, output, error, log):
        self.input = input
        self.output = output
        self.error = error
        self.log = log

    def execute(self):
        print("DateTime " + time.strftime("%c"))
        time.sleep(90) # delays for 1.5 min

        self.log.info("Finished executing the Transform script")


hook = DelayScript(input, output, error, log)
